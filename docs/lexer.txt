Lexical design:
/////////////////

#$&'?@\_` unused symbols

indentifier:     => can be also be a keyword token
	start:       letter, underscore
	followed_by: letter, underscore, number
	until:       not letter, underscore, number or line_end
	until_error: cant error

string:
	start:       "
	followed_by: any u8 character
	until:       "
	until_error: not terminated by '"'

number:
	start:       any number
	followed_by: numbers
	until:       closing 'f' or not number
	until_error: number type specific rule
	notation:    0xF0, 2, 2.0, 2.0f and negative variants

symbol:
	start:       any special symbol
	followed_by: any special symbol
	until:       combination is valid
	until_error  -
